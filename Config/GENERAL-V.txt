#----------------------- 编译参数 -----------------------#
CONFIG_DEVEL=y                          # 启用开发模式（显示更多调试信息、强制选项可见）
CONFIG_CCACHE=y                         # ccache：增量编译缓存，二次编译提速 3~10 倍
CONFIG_TARGET_PER_DEVICE_ROOTFS=y       # 每个机型独立 rootfs；<n> 包可自动变 <m>，方便按需打包，一次性 make，后面单独给不同机型打固件 2~3 min 搞定；
# CONFIG_TARGET_KERNEL_PARTSIZE=128     # 内核分区大小 (MB)，默认 16 MB，够用可不动
# CONFIG_TARGET_ROOTFS_PARTSIZE=1024    # rootfs 分区大小 (MB)，默认 104 MB， ext4 建议 512+
CONFIG_TARGET_ROOTFS_EXT4FS=n           # 禁用 EXT4，仅生成 SquashFS + overlay，省空间
CONFIG_IB=y                             # 生成 ImageBuilder（二次固件打包 SDK）
CONFIG_IB_STANDALONE=y                  # ImageBuilder 自带 feeds，可脱离编译树独立使用
