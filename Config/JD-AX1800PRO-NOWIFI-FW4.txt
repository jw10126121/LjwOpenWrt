# FW4 覆盖文件 — 仅包含与 FW3 基线不同的行

# --- 防火墙栈切换（FW4 nftables）---
CONFIG_PACKAGE_firewall4=y                 # 启用 FW4
CONFIG_PACKAGE_firewall=n                  # 关闭 FW3
CONFIG_PACKAGE_nftables=y                  # nftables 用户态
CONFIG_PACKAGE_kmod-nft-core=y             # nftables 核心
CONFIG_PACKAGE_kmod-nft-nat=y              # nft NAT 核心
CONFIG_PACKAGE_kmod-nft-offload=y          # nft flow offload
CONFIG_PACKAGE_kmod-nft-fullcone=y         # nft FullCone NAT
CONFIG_PACKAGE_kmod-nf-conntrack=y         # 连接跟踪核心
CONFIG_PACKAGE_kmod-nf-conntrack-netlink=y # 连接跟踪 Netlink 接口
CONFIG_PACKAGE_kmod-nf-conntrack6=y        # IPv6 连接跟踪
CONFIG_PACKAGE_iptables=n                  # 关闭 iptables 用户态
CONFIG_PACKAGE_ip6tables=n                 # 关闭 IPv6 iptables
CONFIG_PACKAGE_ip6tables-extra=n           # 关闭 IPv6 iptables 扩展
CONFIG_PACKAGE_ip6tables-mod-nat=n         # 关闭 IPv6 NAT 模块
CONFIG_PACKAGE_iptables-mod-fullconenat=n  # 关闭 iptables FullCone
CONFIG_PACKAGE_kmod-ipt-fullconenat=n      # 关闭 iptables FullCone 内核模块

# --- 代理插件状态切换 ---
CONFIG_PACKAGE_luci-app-ssr-plus=n          # FW4 下关闭 SSR Plus
CONFIG_PACKAGE_luci-i18n-ssr-plus-zh-cn=n   # FW4 下关闭 SSR Plus 中文语言包
CONFIG_PACKAGE_chinadns-ng=n                # SSR Plus 依赖，FW4 下关闭
CONFIG_PACKAGE_ipt2socks=n                  # SSR Plus 依赖，FW4 下关闭
CONFIG_PACKAGE_luci-app-homeproxy=y         # FW4 默认启用 HomeProxy

# --- 其他差异 ---
CONFIG_PACKAGE_luci-app-adguardhome=n        # AdGuard Home，FW4 下关闭
CONFIG_PACKAGE_luci-i18n-adguardhome-zh-cn=n # AdGuard Home 中文语言包
CONFIG_PACKAGE_luci-app-turboacc=n           # FW4 下关闭 Turbo ACC
CONFIG_PACKAGE_luci-i18n-turboacc-zh-cn=n    # Turbo ACC 中文语言包
