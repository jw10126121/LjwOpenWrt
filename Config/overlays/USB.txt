# OVERLAY_GROUP=usb

CONFIG_PACKAGE_kmod-usb-acm=y           # USB ACM 串口
CONFIG_PACKAGE_kmod-usb-serial-option=y # 常见 4G/5G 模块串口
CONFIG_PACKAGE_usbutils=y               # lsusb 等工具
